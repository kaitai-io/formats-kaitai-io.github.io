sub _read {
    my ($self) = @_;

    $self->{b1} = $self->{_io}->read_u1();
    if ($self->b1() == 130) {
        $self->{int2} = $self->{_io}->read_u2be();
    }
}

sub result {
    my ($self) = @_;
    return $self->{result} if ($self->{result});
    $self->{result} = (($self->b1() & 128) == 0 ? $self->b1() : $self->int2());
    return $self->{result};
}
