Usage

Using Kaitai Struct in C++/STL usually consists of 3 steps.

  1. We need to create an STL input stream (std::istream).
    • One can open a stream for reading from a local file: 
      #include <fstream>

std::ifstream is("path/to/local/file.vol", std::ifstream::binary);
    • Or one can prepare a stream for reading from existing std::string str: 
      #include <sstream>

std::istringstream is(str);
    • Or one can parse arbitrary char* buffer in memory, given that we know its size: 
      #include <sstream>

const char buf[] = { ... };
std::string str(buf, sizeof buf);
std::istringstream is(str);
  2. We need to wrap our input stream into Kaitai stream: 
    #include <kaitai/kaitaistream.h>

kaitai::kstream ks(&is);
  3. And finally, we can invoke the parsing: 
    gran_turismo_vol_t data(&ks);

After that, one can get various attributes from the structure by invoking getter methods like:

data.magic() // => get magic

    f_files = false;

void gran_turismo_vol_t::_read() {
    m_magic = m__io->ensure_fixed_contents(std::string("\x47\x54\x46\x53\x00\x00\x00\x00", 8));
    m_num_files = m__io->read_u2le();
    m_num_entries = m__io->read_u2le();
    m_reserved = m__io->ensure_fixed_contents(std::string("\x00\x00\x00\x00", 4));
    int l_offsets = num_files();
    m_offsets = new std::vector<uint32_t>();
    m_offsets->reserve(l_offsets);
    for (int i = 0; i < l_offsets; i++) {
        m_offsets->push_back(m__io->read_u4le());
    }
}

gran_turismo_vol_t::~gran_turismo_vol_t() {
    delete m_offsets;
    if (f_files) {
        for (std::vector<file_info_t*>::iterator it = m_files->begin(); it != m_files->end(); ++it) {
            delete *it;
        }
        delete m_files;
    }
}

}

int32_t gran_turismo_vol_t::file_info_t::size() {
    if (f_size)
        return m_size;
    m_size = ((_root()->offsets()->at((offset_idx() + 1)) & 4294965248) - _root()->offsets()->at(offset_idx()));
    f_size = true;

std::string gran_turismo_vol_t::file_info_t::body() {
    if (f_body)
        return m_body;
    n_body = true;
    if (!(is_dir())) {
        n_body = false;
        std::streampos _pos = m__io->pos();
        m__io->seek((_root()->offsets()->at(offset_idx()) & 4294965248));
        m_body = m__io->read_bytes(size());
        m__io->seek(_pos);
    }
    f_body = true;
    return m_body;
}

bool gran_turismo_vol_t::file_info_t::is_dir() {
    if (f_is_dir)
        return m_is_dir;
    m_is_dir = (flags() & 1) != 0;
    f_is_dir = true;
    return m_is_dir;
}

bool gran_turismo_vol_t::file_info_t::is_last_entry() {
    if (f_is_last_entry)
        return m_is_last_entry;
    m_is_last_entry = (flags() & 128) != 0;
    f_is_last_entry = true;
    return m_is_last_entry;

uint32_t gran_turismo_vol_t::ofs_dir() {

    m_ofs_dir = offsets()->at(1);
    f_ofs_dir = true;
    return m_ofs_dir;
}

std::vector<gran_turismo_vol_t::file_info_t*>* gran_turismo_vol_t::files() {
    if (f_files)
        return m_files;
    std::streampos _pos = m__io->pos();
    m__io->seek((ofs_dir() & 4294965248));
    int l_files = _root()->num_entries();
    m_files = new std::vector<file_info_t*>();
    m_files->reserve(l_files);
    for (int i = 0; i < l_files; i++) {
        m_files->push_back(new file_info_t(m__io, this, m__root));
    }
    m__io->seek(_pos);
    f_files = true;
    return m_files;
}
