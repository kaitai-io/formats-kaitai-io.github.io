    pub fn value(
        &self

        let _io = self._io.borrow();

        let _prc = self._parent.get_value().borrow().upgrade();
        let _r = _rrc.as_ref().unwrap();

            return Ok(self.value.borrow());

        Ok(self.value.borrow())
    }
