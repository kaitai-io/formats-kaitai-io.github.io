sub _read {
    my ($self) = @_;

    $self->{most_significant_bits} = $self->{_io}->read_u2le();
    $self->{least_significant_bits} = $self->{_io}->read_u2le();
}

sub value {
    my ($self) = @_;
    return $self->{value} if ($self->{value});
    $self->{value} = ($self->least_significant_bits() + ($self->most_significant_bits() << 16));
    return $self->{value};
}
