OpenPGP message: Ruby parsing library

The OpenPGP Message Format is a format to store encryption and signature keys for emails.

File extension

["gpg", "pub", "pgp"]

Usage

Parse a local file and get structure in memory:

data = OpenpgpMessage.from_file("path/to/local/file.["gpg", "pub", "pgp"]")

Or parse structure from a string of bytes:

bytes = "\x00\x01\x02..."
data = OpenpgpMessage.new(Kaitai::Struct::Stream.new(bytes))

After that, one can get various attributes from the structure by invoking getter methods like:

data.packets # => get packets

      when :packet_tags_user_id_packet
        @_raw_body = @_io.read_bytes(len)
        io = Kaitai::Struct::Stream.new(@_raw_body)
        @body = UserIdPacket.new(io, self, @_root)
      when :packet_tags_secret_key_packet
        @_raw_body = @_io.read_bytes(len)
        io = Kaitai::Struct::Stream.new(@_raw_body)
        @body = SecretKeyPacket.new(io, self, @_root)
      when :packet_tags_signature_packet
        @_raw_body = @_io.read_bytes(len)
        io = Kaitai::Struct::Stream.new(@_raw_body)
        @body = SignaturePacket.new(io, self, @_root)
      when :packet_tags_public_subkey_packet
        @_raw_body = @_io.read_bytes(len)
        io = Kaitai::Struct::Stream.new(@_raw_body)
        @body = PublicKeyPacket.new(io, self, @_root)
      else
        @body = @_io.read_bytes(len)
      end
      self
    end
    attr_reader :len
    attr_reader :body
    attr_reader :_raw_body
  end

  class LenSubpacket < Kaitai::Struct::Struct
    def initialize(_io, _parent = nil, _root = self)
      super(_io, _parent, _root)
      _read
    end

    def _read
      @first_octet = @_io.read_u1
      if  ((first_octet >= 192) && (first_octet < 255)) 
        @second_octet = @_io.read_u1
      end
      if first_octet == 255
        @scalar = @_io.read_u4be
      end
      self
    end
    def len
      return @len unless @len.nil?
      @len = (first_octet < 192 ? first_octet : ( ((first_octet >= 192) && (first_octet < 255))  ? ((((first_octet - 192) << 8) + second_octet) + 192) : scalar))
      @len
    end
    attr_reader :first_octet
    attr_reader :second_octet
    attr_reader :scalar
  end

  class Packet < Kaitai::Struct::Struct
    def initialize(_io, _parent = nil, _root = self)
      super(_io, _parent, _root)
      _read
    end

    def _read
      @one = @_io.read_bits_int(1) != 0
      @new_packet_format = @_io.read_bits_int(1) != 0
      if new_packet_format
        @packet_type_new = Kaitai::Struct::Stream::resolve_enum(PACKET_TAGS, @_io.read_bits_int(6))
      end
      if !(new_packet_format)
        @packet_type_old = Kaitai::Struct::Stream::resolve_enum(PACKET_TAGS, @_io.read_bits_int(4))
      end
      if !(new_packet_format)
        @len_type = @_io.read_bits_int(2)
      end
      @_io.align_to_byte
      case new_packet_format
      when false
        @body = OldPacket.new(@_io, self, @_root)
      end
      self
    end
    attr_reader :one
    attr_reader :new_packet_format
    attr_reader :packet_type_new
    attr_reader :packet_type_old
    attr_reader :len_type
    attr_reader :body
  end
