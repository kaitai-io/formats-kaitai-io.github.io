DOS MZ executable: PHP parsing library

DOS MZ file format is a traditional format for executables in MS-DOS environment. Many modern formats (i.e. Windows PE) still maintain compatibility stub with this format.

As opposed to .com file format (which basically sports one 64K code segment of raw CPU instructions), DOS MZ .exe file format allowed more flexible memory management, loading of larger programs and added support for relocations.

class DosMz extends \Kaitai\Struct\Struct {
    public function __construct(\Kaitai\Struct\Stream $_io, \Kaitai\Struct\Struct $_parent = null, \DosMz $_root = null) {
        parent::__construct($_io, $_parent, $_root);
        $this->_read();
    }

    private function _read() {
        $this->_m_hdr = new \DosMz\MzHeader($this->_io, $this, $this->_root);
        $this->_m_mzHeader2 = $this->_io->readBytes(($this->hdr()->relocationsOfs() - 28));
        $this->_m_relocations = [];
        $n = $this->hdr()->qtyRelocations();
        for ($i = 0; $i < $n; $i++) {

        }
        $this->_m_body = $this->_io->readBytesFull();
    }
