proc len(this: QuicktimeMov_Atom): int = 
  if this.lenInstFlag:
    return this.lenInst
  let lenInstExpr = int((if this.len32 == 0: this.io.size - 8 else: (if this.len32 == 1: this.len64 - 16 else: this.len32 - 8)))
  this.lenInst = lenInstExpr
  this.lenInstFlag = true
  return this.lenInst
