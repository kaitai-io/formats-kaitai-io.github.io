sub _read {
    my ($self) = @_;

    $self->{entries} = ();
    while (!$self->{_io}->is_eof()) {
        push @{$self->{entries}}, Ipv4Packet::Ipv4Option->new($self->{_io}, $self, $self->{_root});
    }
}
